\begin{lstlisting}
def desenha_quadrado( tamanho, cor ):
    img = pygame.Surface( tamanho )

    cinza = Color( "#808080" )
    cor1 = pygame.color.add( cor, cinza )
    cor2 = pygame.color.subtract( cor, cinza )

    r = Rect( ( 0, 0 ), tamanho )
    r.inflate_ip( -2, -2 )
    r.topleft = ( 1, 1 )

    img.fill( cor )

    line = pygame.draw.line
    line( img, cor1, r.topleft, r.topright )
    line( img, cor1, r.topleft, r.bottomleft )
    line( img, cor2, r.bottomleft, r.bottomright )
    line( img, cor2, r.bottomright, r.topright )

    return img
# desenha_quadrado()
\end{lstlisting}

\frame[containsverbatim]{
  \frametitle{Objeto do jogo ``Quadrado''}

\begin{lstlisting}
class Quadrado( Sprite ):
    grupos = None # Grupos ao qual este sprite pertence
                  # depois esta variavel de classe sera atribuida
                  # para lembrar dos objetos nao desenhados
    tamanho = ( 50, 50 )
    def __init__( self, pos=( 0, 0 ) ):
        Sprite.__init__( self, self.grupos )
        self.rect = Rect( pos, self.tamanho )
    # __init__()
# Quadrado
\end{lstlisting}

\begin{itemize}
  \item \textbf{Nunca se esqueça de chamar
    \lstinline|Sprite.__init__|!}
  \item \lstinline|Sprite.__init__| pode receber uma lista de grupos a
  qual este Sprite pertence. Nós usaremos o atributo de
  \textbf{classe}\footnote{Apesar de usarmos \lstinline|self.grupos|,
  isso foi feito devido ao método de resolução de atributos do Python,
  primeiro procura-se na instância, depois nas classes que compõe a
  instância.}, que será atribuído mais tarde, assim todos os
  Sprites se adicionam automaticamente ao grupo correto.
\end{itemize}
}

\frame[containsverbatim]{
  \frametitle{Lendo imagens para o labirinto}
  \begin{itemize}
  \item Usamos variáveis de classe para evitar que cada instância
    carrege uma cópia, economizando memória
  \item Se precisarmos mudar de todas as instâncias, é só mudar esta
    variável de classe.
  \item Se quisermos que uma certa instância tenha uma imagem
    diferente, é só atribuir à instância e este valor se sobrepõe ao
    da classe na resolução dos atributos. (Maior Flexibilidade)
  \end{itemize}
}


\frame[containsverbatim]{
  \frametitle{Desenhando o Labirinto}

\begin{lstlisting}
    def desenhe( self, tudo=False ):
        tela = self.tela
        nao_desenhados = self.nao_desenhados
        if tudo:
            for l in self.mapa:
                for p in l:
                    tela.blit( p.image, p )
        else:
            nao_desenhados.draw( self.tela )

        nao_desenhados.empty()
    # desenhe()
\end{lstlisting}

\begin{itemize}
\item O uso de sprites e grupos facilita operações como desenhar
  apenas o que mudou.
\item Uma instância de Quadrado, quando criada, vai automaticame para
  o grupo de não desenhados, pois atribuímos este grupo a
  \lstinline|Quadrado.grupos|.
\item Como sempre desenhamos todos os quadrados não desenhados, após a
  operação esvaziamos o grupo.
\end{itemize}
}

class Jogo( object ):
    FPS = 24
    RESOLVE_PASSOS_POR_SEG = 0

    def __init__( self, mapa, tamanho=( 800, 600 ) ):
        self.clock = pygame.time.Clock()
        self.tela  = pygame.display.set_mode( tamanho )
        self.mapa  = mapa
        self.le_labirinto()
    # __init__()


    def le_labirinto( self ):
        self.labirinto = Labirinto( self.tela, self.mapa )
    # le_labirinto()

\frame[containsverbatim]{
  \frametitle{O Jogo: Parando e Atualizando a tela}

\begin{lstlisting}
    def termine( self ):
        raise StopIteration
    # termine()


    def atualize( self, tudo=False ):
        if tudo:
            pygame.display.flip()
        else:
            pygame.display.update()
    # atualize()
\end{lstlisting}

\begin{itemize}
\item Note que devido à falta de argumentos para
  \lstinline|pygame.display.update()|, este tem o mesmo efeito
  que \lstinline|pygame.display.flip()|: atualiza a tela inteira.
\item Em uma implementação futura poderíamos lembrar as áreas que
  mudaram, resultantes do \lstinline|nao_desenhados.draw()| e passar
  como argumento para o \lstinline|pygame.display.update()|,
  economizando ciclos do processador.
\end{itemize}

}

    def trata_eventos( self ):
        for e in pygame.event.get( [ KEYDOWN, QUIT, ACTIVEEVENT ] ):
            if   e.type == QUIT:
                self.termine()

            elif e.type == KEYDOWN:
                if   e.key == K_F2:
                    self.le_labirinto()

                elif e.key == K_F3:
                    def callback( estagio ):
		        pass # removido para caber no slide
                    # callback()
                    self.labirinto.resolve( callback )
                    self.labirinto.desenhe()
                    self.atualize()

                elif e.key == K_ESCAPE:
                    self.termine()

            elif e.type == ACTIVEEVENT:
                self.labirinto.desenhe( True )

        try:
            # ActiveEvent faz desenhar a tela de novo
            pygame.event.post( pygame.event.Event( ACTIVEEVENT ) )
